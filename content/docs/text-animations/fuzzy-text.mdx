## Examples

```tsx
// White text on black background
<FuzzyText
  text="STATIC NOISE"
  baseColor="#ffffff"
  glowColor="#ffffff"
  enableHover={true}
/>

// Green terminal style
<FuzzyText
  text="ACCESS DENIED"
  baseColor="#00ff00"
  glowColor="#00ff00"
  noiseIntensity={1.2}
/>

// Red error/warning
<FuzzyText
  text="404 NOT FOUND"
  baseColor="#ff0040"
  glowColor="#ff0040"
  enableHover={false}
/>

// Subtle effect
<FuzzyText
  text="Subtle Glitch"
  noiseIntensity={0.5}
  speed={80}
/>

// Fast intense noise
<FuzzyText
  text="CHAOS"
  noiseIntensity={2}
  speed={20}
/>
```
