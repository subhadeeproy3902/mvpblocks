## Props

<TypeTable
  type={{
    text: {
      description: "The final text to reveal after the scramble animation.",
      type: "string",
      default: "required",
    },
    trigger: {
      description: "When to trigger the animation. 'scroll' ties reveal progress to scroll position.",
      type: "'mount' | 'hover' | 'scroll'",
      default: "'mount'",
    },
    scrambleSpeed: {
      description: "How fast the random characters shuffle (in ms). Lower = faster scramble.",
      type: "number",
      default: "50",
    },
    revealDelay: {
      description: "Delay between each character being revealed (in ms). Higher = slower decode.",
      type: "number",
      default: "100",
    },
    className: {
      description: "Optional CSS class for styling.",
      type: "string",
      default: "''",
    },
    characterSet: {
      description: "Custom characters to use for the scramble effect.",
      type: "string",
      default: "'ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789!@#$%^&*'",
    },
    sequential: {
      description: "If true, characters reveal left-to-right. If false, characters randomly lock in place.",
      type: "boolean",
      default: "true",
    },
    onComplete: {
      description: "Callback function fired when the animation completes.",
      type: "() => void",
      default: "undefined",
    },
  }}
/>

## Speed Examples

```tsx
// Slow dramatic decode (good for hero sections)
<ShuffleText text="Welcome" scrambleSpeed={60} revealDelay={200} />

// Fast decode (good for hover effects)
<ShuffleText text="Click Me" scrambleSpeed={30} revealDelay={80} />

// Ultra slow cinematic decode
<ShuffleText text="Loading..." scrambleSpeed={80} revealDelay={300} />
```
